                pageNo = 1;

                $('div.ui-datepicker-title span.ui-datepicker-year').text(date.substr(0, 4));
                $('div.ui-datepicker-title span.month').text(date.substr(4, 2));

                getNewsList();
            });

            $('img.ui-datepicker-trigger').on('click', function () {
                if (calendarContainer.find('table').length > 0) {
                    assignEvent();
                }

                function assignEvent() {
                    var _currentYear = parseInt(date.substr(0, 4));
                    var _currentMonth = date.substr(4, 2);

                    dateSelect.find('input[type="text"]').datepicker('setDate', dateObj);

                    calendarContainer.find('tbody td').each(function () {
                        $(this).data('year', _currentYear);
                        $(this).data('month', _currentMonth);
                    });

                    calendarContainer.find('tbody td a').on('click', function () {
                        date = $(this).closest('td').data('year') + $(this).closest('td').data('month') + feUI.zeroFill($(this).text(), 2);

                        dateObj.setFullYear(parseInt(date.substr(0, 4)));
                        dateObj.setMonth(parseInt(date.substr(4, 2)) - 1);
                        dateObj.setDate(parseInt(date.substr(6)));
                        pageNo = 1;

                        getNewsList();
                    });

                    calendarContainer.find('div.ui-datepicker-header a').each(function (idx) {
                        $(this).on('click', function () {
                            dateObj.setMonth(dateObj.getMonth() + (idx === 0? -1: 1));

                            date = feUI.makeDateString(dateObj, true);

                            assignEvent();
                        });
                    });
                }
            });

            getNewsList();

            function getNewsList() {
                dateSelect.find('strong.date-text').text(date.substr(0, 4) + '.' + date.substr(4,
                    2) + '.' + date.substr(6));

                var _url = SR + 'news/getNewsList?p=' + pageNo + '&ct=6&srt=';

                _url += sort.filter(':checked').val() + '&d=' + date;

                $.ajax({
                    type: 'get',
                    url: _url,
                    contentType: 'application/json',
                    dataType: 'json'
                }).done(function (d) {
                    if (d && d.list && d.list.length > 0) {
                        var _html = '';
                        var _link, _date, _img, _temp, _content;

                        d.list.forEach(function (item) {
                            _temp = newsListTemplate;
                            _link = SR + 'news/' + item.sequence + '?srt=' + sort.filter(':checked').val() + '&d=Y';
                            _date = new Date(item.createdAt);
                            _img = (item.preview)? item.preview: '/asset/img/etc/default_main_thumb.jpg';

                            if (item.type !== 'video') {
                                _temp = newsListTemplate.replace(/<i class="icon-play"><\/i>/gi, '');
                            }
                            _content = item.content.replace(/<div[^>]*class=["']?ynaobject["']?[^>]*>.*?<\/div>/gs, '').replace(/(<[a-zA-z]+((?!>).)*>)/gi, '').replace(/<(\/[a-zA-Z]+)>/gi,'');
